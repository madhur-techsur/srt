Replace the stub `RequestList.tsx` (which currently returns null) with a full implementation. The component MUST:

1. **Imports** — at the top of the file:
   ```typescript
   import { useEffect, useState } from 'react';
   import { getRequests } from '../api/requestsApi';
   import { Request } from '../types/request';
   ```

2. **State** — three state variables:
   ```typescript
   const [requests, setRequests] = useState<Request[]>([]);
   const [loading, setLoading] = useState<boolean>(true);
   const [error, setError] = useState<string | null>(null);
   ```

3. **useEffect on mount** — fetch requests when component mounts (key prop change from App.tsx causes remount, which re-fires this effect — NO additional prop or callback needed):
   ```typescript
   useEffect(() => {
     setLoading(true);
     setError(null);
     getRequests()
       .then((data) => {
         setRequests(data);
       })
       .catch(() => {
         setError('Could not load requests.');
       })
       .finally(() => {
         setLoading(false);
       });
   }, []);
   ```

4. **Render logic** — in JSX return:
   - If `loading === true`: render a loading indicator (e.g., `<p>Loading...</p>`) — optional but expected by Playwright test
   - If `error !== null`: render `<p data-testid="error-state">{error}</p>` — shows "Could not load requests."
   - If `requests.length === 0` and not loading and no error: render `<p data-testid="empty-state">No requests yet.</p>`
   - Otherwise: render table:
     ```tsx
     <table data-testid="request-table">
       <thead>
         <tr>
           <th>Name</th>
           <th>Title</th>
           <th>Description</th>
         </tr>
       </thead>
       <tbody>
         {requests.map((req) => (
           <tr key={req.id}>
             <td>{req.name}</td>
             <td>{req.title}</td>
             <td>{req.description}</td>
           </tr>
         ))}
       </tbody>
     </table>
     ```

5. **data-testid attributes** — REQUIRED for Playwright selectors:
   - `data-testid="request-table"` on the `<table>` element
   - `data-testid="empty-state"` on the empty-state `<p>` element
   - `data-testid="error-state"` on the error-state `<p>` element

6. **Columns to render**: ONLY `name`, `title`, `description`. Do NOT render `id` or `createdAt` — demo scope.

7. **TypeScript interface** already in `frontend/src/types/request.ts`:
   ```typescript
   export interface Request {
     id: number;
     name: string;
     title: string;
     description: string;
     createdAt: string;
   }
   ```
   Use the `Request` type — do NOT use `any`.

8. **getRequests()** already in `frontend/src/api/requestsApi.ts`:
   ```typescript
   export async function getRequests(): Promise<Request[]> {
     const res = await fetch(`${BASE_URL}/api/requests`, {
       headers: { 'Accept': 'application/json' },
     });
     if (!res.ok) {
       const error: ApiError = await res.json();
       throw error;
     }
     return res.json() as Promise<Request[]>;
   }
   ```
   Call `getRequests()` directly — do NOT re-implement the fetch.

9. **Auto-refresh mechanism** — App.tsx already has:
   ```tsx
   const [refreshKey, setRefreshKey] = useState(0);
   const handleSuccess = () => setRefreshKey((k) => k + 1);
   // ...
   <RequestList key={refreshKey} />
   ```
   When `refreshKey` increments, React remounts RequestList entirely, causing `useEffect` to re-fire. No additional prop or callback is needed in RequestList. The `key` prop is on `<RequestList>` in App.tsx — do NOT modify App.tsx.

- Backend must be running at `localhost:8080` during tests (either started manually or in CI via separate webServer config). Add a `baseURL` to point to frontend at `localhost:5173`.
- For SC2 empty-state test: Since H2 is in-memory and starts empty each boot, consider ordering the empty-state test FIRST (before any data is submitted) using `test.describe` ordering, or mock the API response.
- Use `page.waitForSelector` or `await expect(locator).toBeVisible()` for async assertions.
- Each test should use unique test data to avoid cross-test contamination.

**Exact test structure:**

```typescript
import { test, expect } from '@playwright/test';

test.describe('RequestList', () => {
  test('SC2: shows empty state when no requests exist', async ({ page }) => {
    // This test should run when backend has no data (first run, fresh H2)
    await page.goto('/');
    // Wait for loading to complete
    await page.waitForSelector('[data-testid="empty-state"], [data-testid="request-table"]');
    // If no data, empty state should show
    const emptyState = page.locator('[data-testid="empty-state"]');
    const requestTable = page.locator('[data-testid="request-table"]');
    // Assert: either empty state is visible OR table is visible (depends on backend state)
    // For strict SC2 test: check that empty-state shows correct text when visible
    const emptyStateVisible = await emptyState.isVisible();
    if (emptyStateVisible) {
      await expect(emptyState).toHaveText('No requests yet.');
      await expect(requestTable).not.toBeVisible();
    }
  });

  test('SC3: table auto-refreshes after form submission without page reload', async ({ page }) => {
    await page.goto('/');
    // Fill in form
    const uniqueName = `Test User ${Date.now()}`;
    await page.fill('[data-testid="name-input"]', uniqueName);
    await page.fill('[data-testid="title-input"]', 'Auto-refresh Test Title');
    await page.fill('[data-testid="description-input"]', 'Testing auto-refresh behavior');
    // Submit form
    await page.click('[data-testid="submit-button"]');
    // Wait for success acknowledgment (from Phase 2 RequestForm)
    await page.waitForSelector('[data-testid="success-message"]', { timeout: 5000 });
    // Assert table appears and contains the new record — no page.reload() called
    await expect(page.locator('[data-testid="request-table"]')).toBeVisible({ timeout: 5000 });
    await expect(page.locator('[data-testid="request-table"]')).toContainText(uniqueName);
  });

  test('SC1: table displays stored requests with Name, Title, Description columns', async ({ page }) => {
    // POST a request directly via API to ensure there is data
    const response = await page.request.post('http://localhost:8080/api/requests', {
      data: {
        name: 'SC1 Test User',
        title: 'SC1 Test Title',
        description: 'SC1 Test Description',
      },
      headers: { 'Content-Type': 'application/json' },
    });
    expect(response.status()).toBe(201);

    await page.goto('/');
    // Wait for table to appear
    await expect(page.locator('[data-testid="request-table"]')).toBeVisible({ timeout: 5000 });
    // Assert column headers
    await expect(page.locator('[data-testid="request-table"] thead th').nth(0)).toHaveText('Name');
    await expect(page.locator('[data-testid="request-table"] thead th').nth(1)).toHaveText('Title');
    await expect(page.locator('[data-testid="request-table"] thead th').nth(2)).toHaveText('Description');
    // Assert data row contains submitted values
    await expect(page.locator('[data-testid="request-table"] tbody')).toContainText('SC1 Test User');
    await expect(page.locator('[data-testid="request-table"] tbody')).toContainText('SC1 Test Title');
  });
});
```
